    modifier delegatedOwner(
        bytes memory data,
        uint256 signatureExpiresAt,
        bytes memory sig
    ) {
        if (msg.sender != owner()) {
            require(now < signatureExpiresAt, "Signature expired");
 
            bytes32 operationHash = keccak256(abi.encodePacked(
                data,
                signatureExpiresAt
            ));
 
            // This checks that the hash hasn't been used before
            _replayGuard(operationHash);
 
            require(
                getSigner(operationHash, sig) == owner(),
                "Not owner or invalid signature"
            ); 
        }
        _;
    }

    /**
     * @dev Gets the next timestamp in which the subscription can be executed.
     * It is possible for this method to return a timestamp in the past, so
     * publishers need to provide a grace period because of the delay in
     * transaction execution.
     * @param subscriptionHash The hash of the subscription.
     * @return The timestamp in which the subscription can be executed.

    function nextPaymentTimestamp(bytes32 subscriptionHash)
        public
        view
        returns(uint256)
    {
        Subscription memory sub = _subscriptions[subscriptionHash];
 
        // Periods from the start of the subscription for the next payment
        uint256 nextPeriods = sub.executions.mul(sub.periods);
 
        if (sub.periodUnit == PeriodUnit.Day) {
            return sub.createdAt.addDays(nextPeriods);
        }
        
        if (sub.periodUnit == PeriodUnit.Month) {
            return sub.createdAt.addMonths(nextPeriods);
        }
        
        Eif (sub.periodUnit == PeriodUnit.Year) {
            return sub.createdAt.addYears(nextPeriods);
        }
 
        // We should never get here
        assert(false);
    }

    /**
     * @dev Executes the subscription (transfers tokens) or deletes it.
     * The subscription gets deleted when:
     *   1) the user doesnt have enough funds
     *   2) it has already reached its max executions
     *   3) the state is invalid
     *   4) the subscription was cancelled.
     * @param subscriptionHash The hash of the subscription to execute.
     */
    function execute(bytes32 subscriptionHash) public returns(bool) {
        Subscription storage sub = _subscriptions[subscriptionHash];
 
        // Emit event if the subscription doesn't exist
        if(!_subscriptionExists(subscriptionHash)) {
            emit SubscriptionNotFound(subscriptionHash);
            return false;
        }
 
        uint256 nextPayment = nextPaymentTimestamp(subscriptionHash);
 
        // Emit event if next payment timestamp hasn't been reached
        if (now < nextPayment) {
            emit SubscriptionNotReady(subscriptionHash, nextPayment);
            return false;
        }
 
        // Delete if subscription was cancelled
        if (isCancelled(subscriptionHash)) {
            _delete(subscriptionHash, DeletionReason.CANCELLED);
            return false;
        }
 
        // Delete if subscription already reached maxExecutions
        if (sub.maxExecutions != 0 && sub.executions >= sub.maxExecutions) {
            _delete(subscriptionHash, DeletionReason.EXPIRED);
            return false;
        }
 
        // Delete if state is invalid
        if (!_validateSubscriptionState(subscriptionHash, sub)) {
            _delete(subscriptionHash, DeletionReason.INVALID_STATE);
            return false;
        }
 
        IERC20 token = IERC20(sub.token);
        address subscriber = sub.subscriber;
        uint256 amount = sub.amount;
 
        bool hasEnoughFunds = (
            token.balanceOf(subscriber) >= amount &&
            token.allowance(subscriber, address(this)) >= amount
        );
 
        // If user doesn't have enough funds, delete the subscription
        if (!hasEnoughFunds) {
            _delete(subscriptionHash, DeletionReason.NOT_ENOUGH_FUNDS);
            return false;
        }
 
        // It is not possible to overflow a uint256 with increments of one
        sub.executions += 1;
 
        emit SubscriptionExecuted(subscriptionHash, nextPayment);
 
        // Execute transfer
        // This call shouldn't revert if the token actually follows ERC20
        token.safeTransferFrom(subscriber, wallet, amount);
 
        return true;
    }

    /**
     * @dev Internal function that deletes a subscription. It emits an event
     * with the deletion reason.
     * @param subscriptionHash The hash of the subscription.
     * @param reason The reason the subscription got deleted.
     */
    function _delete(
        bytes32 subscriptionHash,
        DeletionReason reason
    )
        internal
    {
        Erequire(
            _subscriptionExists(subscriptionHash),
            "Subscription doesn't exist"
        );
 
        address subscriber = _subscriptions[subscriptionHash].subscriber;
 
        // Remove from subscriber's array
        _removeFromArrayAndSwap(
            _subscriptionsBySubscriber[subscriber],
            _subscriptionSubscriberIndex,
            subscriptionHash
        );
 
        // Remove from all subscriptions array
        _removeFromArrayAndSwap(
            _allSubscriptions,
            _allSubscriptionsIndex,
            subscriptionHash
        );
 
        // Set all subscription fields to 0
        delete _subscriptions[subscriptionHash];
 
        _deleted[subscriptionHash] = true;
 
        emit SubscriptionDeleted(subscriptionHash, reason);
    }
 
    /**
     * @dev Internal function that creates a subscription.
     * The base implementation doesn't expose functions to create
     * subscriptions, so this must be used by child contracts to do so.
     * @param subscriptionHash The hash of the subscription. This is a
     * parameter because different usecases might hash the subscription
     * differently.
     * @param subscriber The address of the subscriber.
     * @param token The address of the token used for payments.
     * @param amount The amount of tokens transferred in each payment.
     * @param periodUnit Unit to use for periods (Day, Month, Year).
     * @param periods Number period units between each payment.
     * @param maxExecutions Number of times the subscription can be executed.
     */
    function _create(
        bytes32 subscriptionHash,
        address subscriber,
        address token,
        uint256 amount,
        PeriodUnit periodUnit,
        uint256 periods,
        uint256 maxExecutions
    )
        internal
    {
        require(
            !_subscriptionExists(subscriptionHash) &&
            !_deleted[subscriptionHash],
            "Subscription hash already used"
        );
 
        require(
            subscriber != address(0),
            "Subscriber can't be the zero address"
        );
 
        require(periods != 0, "Periods can't be 0");
        require(amount != 0, "Amount can't be 0");
 
        _subscriptions[subscriptionHash] = Subscription(
            subscriber,
            token,
            amount,
            periodUnit,
            periods,
            maxExecutions,
            0, // executions
            now // createdAt
        );
 
        // Add subscription to subscriber's array
        bytes32[] storage subs = _subscriptionsBySubscriber[subscriber];
        _subscriptionSubscriberIndex[subscriptionHash] = subs.length;
        subs.push(subscriptionHash);
 
        // Add subscription to allSubscriptions array
        _allSubscriptionsIndex[subscriptionHash] = _allSubscriptions.length;
        _allSubscriptions.push(subscriptionHash);
 
        emit SubscriptionCreated(
            subscriptionHash
        );
 
        execute(subscriptionHash);
    }

    /**
     * @dev Private function to remove a subscription from an array.
     * Adapted from OpenZeppelin ERC721 implementation!
     * This has O(1) time complexity, but alters the order of the array.
     * @param array The array to remove the subscription from.
     * @param indexes A storage mapping that maps subscription hashes to
     * indexes in the array.
     * @param item The subscription hash to remove.
     */
    function _removeFromArrayAndSwap(
        bytes32[] storage array,
        mapping(bytes32 => uint256) storage indexes,
        bytes32 item
    )
        private
    {
        // To prevent a gap in the array, we store the last item in the index
        // of the item to delete, and then delete the last slot (swap and pop).
        uint256 lastIndex = array.length.sub(1);
        uint256 index = indexes[item];
 
        // If the subscription being deleted is the last in the array, this
        // is not necessary.
        if (index != lastIndex) {
            bytes32 lastItem = array[lastIndex];
 
            // Move the last item to the slot of the to-delete item
            array[index] = lastItem;
            // Update the moved item's index
            indexes[lastItem] = index;
        }
 
        // This also deletes the contents at the last position of the array
        array.length--;
        indexes[item] = 0;
    }
 
    /**
     * @dev Paginates a bytes32 array.
     * @param offset The offset in which the count begins.
     * @param limit Maximum amount of elements to return.
     */
    function _paginate(
        bytes32[] storage array,
        uint256 offset,
        uint256 limit
    )
        private
        view
        returns(bytes32[] memory items)
    {
        if (array.length <= offset) {
            return new bytes32[](0);
        }
 
        uint256 len = Math.min(limit, array.length - offset);
 
        items = new bytes32[](len);
 
        for (uint256 i = 0; i < len; i++) {
            items[i] = array[i + offset];
        }
    }
