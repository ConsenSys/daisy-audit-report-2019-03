    struct Subscription {
        // Address of the subscriber
        address subscriber;
        // Address of the token
        address token;
        // Number of tokens to transfer
        uint256 amount;
        // Unit used for billing periods
        PeriodUnit periodUnit;
        // Number of PeriodUnits between payments
        uint256 periods;
        // Number of times that it can be executed (0 means no limit)
        uint256 maxExecutions;
        // Timestamp (seconds)
        uint256 start;
 
        // Number of times it has been executed
        uint256 executions;
        // Timestamp (seconds)

    }
 
    // EIP712 Domain hash

        0x035aff83d86937d35b32e04f0ddc6ff469290eef2f1b692d8a815c89404d4749;

    /**
     * @dev The method can only be called by the owner or by providing
     * the owner's signature.
     * @param data Arbitrary data whose hash was signed by the owner.
     * @param sig Signature generated by the owner for the hashed data.
     */
    modifier delegatedOwner(bytes memory data, bytes memory sig) {
        if (msg.sender != owner()) {
            bytes32 operationHash = keccak256(data);
 
            // Replay guard, see DelegatedSubscriptionManager.sol
            _replayGuard(operationHash);
 
            require(
                getSigner(operationHash, sig) == owner(),
                "Not owner or invalid signature"
            ); 
        }
        _;
    }

    function isCancelled(bytes32 subscriptionHash)

    /**
     * @dev Returns if the subscription is considered active, which means that
     * it has been executed at least once and the next payment is in the
     * future. A subscription that is cancelled is still considered active
     * until the end of the period.
     * @param subscriptionHash The hash of the subscription.
     * @return True if the subscription is active, false otherwise.
     */
    function isActive(bytes32 subscriptionHash)
        external
        view
        returns(bool)
    {
        return (
            // Payment in the future
            now < _nextPayment[subscriptionHash] &&
            // Has been executed
            _subscriptions[subscriptionHash].executions != 0
        );
    }
 
    /**
     * @dev Gets the next timestamp in which the subscription can be executed.
     * It is possible for this method to return a timestamp in the past, so
     * publishers need to provide a grace period because of the delay in
     * transaction execution.
     * @param subscriptionHash The hash of the subscription.
     * @return The timestamp in which the subscription can be executed.
     */
    function nextPaymentTimestamp(bytes32 subscriptionHash)
        external
        view
        returns(uint256)
    {
        if (_cancelled[subscriptionHash]) {
            return 0;
        }
 
        return _nextPayment[subscriptionHash];
    }

    /**
     * @dev Cancels a subscription (it can't be executed).
     * @param subscriptionHash The id of the subscription.
     * @param signature Signature generated by subscriber or owner.
     */
    function cancel(
        bytes32 subscriptionHash,
        bytes calldata signature
    )
        external
    {
        bytes32 cancelHash = getActionHash("cancel", subscriptionHash);
        address signer = getSigner(cancelHash, signature);
        address subscriber = _subscriptions[subscriptionHash].subscriber;
        require(
            signer == subscriber || signer == owner(),
            "Invalid signature"
        );
        _cancel(subscriptionHash, false);
    }

            msg.sender == owner(),

    /**
     * @dev Gets the signer for a given hash and signature (using EIP712).
     * @param hash The hash that was signed.
     * @param signature The signature for the given hash.
     * @return The address recovered from the hash and the signature.
     */
    function getSigner(bytes32 hash, bytes memory signature)
        public
        view
        returns(address)
    {
        bytes32 digest = keccak256(abi.encodePacked(
            "\x19\x01",
            domainSeparator,
            hash
        ));
 
        return digest.recover(signature);
    }
 
    /**
     * @dev Executes the subscription (transfers tokens).
     * @param subscriptionHash The hash of the subscription to execute.
     */
    function execute(bytes32 subscriptionHash) public {
        Subscription storage sub = _subscriptions[subscriptionHash];
 
        // Periods is 0 if the subscription doesn't exist,
        // and is set to 0 when a subscription is cancelled.
        require(
            sub.periods != 0,
            "Subscription doesn't exist or was cancelled"
        );
 
        // This prevents reentrancy, as nextPaymentTimestamp allways increases
        require(
            _nextPayment[subscriptionHash] <= now,
            "Subscription not ready"
        );
 
        // Perform state checks (view function)
        _validateSubscriptionState(subscriptionHash, sub);
 
        // It is not possible to overflow a uint256 with increments of one
        sub.executions += 1;
 
        // nextPaymentTimestamp always increases as sub.periods can't be 0.
        uint256 nextPeriods = sub.executions.mul(sub.periods);
 
        if (sub.periodUnit == PeriodUnit.Day) {
            _nextPayment[subscriptionHash] = sub.start.addDays(nextPeriods);
        } else if (sub.periodUnit == PeriodUnit.Month) {
            _nextPayment[subscriptionHash] = sub.start.addMonths(nextPeriods);
        } else Eif (sub.periodUnit == PeriodUnit.Year) {
            _nextPayment[subscriptionHash] = sub.start.addYears(nextPeriods);
        }
 
        IERC20 token = IERC20(sub.token);
        address subscriber = sub.subscriber;
 
        bool hasEnoughFunds = (
            token.balanceOf(subscriber) >= sub.amount &&
            token.allowance(subscriber, address(this)) >= sub.amount
        );
 
        if (hasEnoughFunds) {
            // Execute transfer
            token.safeTransferFrom(
                sub.subscriber,
                wallet,
                sub.amount
            );
 
            emit SubscriptionExecuted(subscriptionHash);
        }
 
        bool expired = (
            sub.maxExecutions != 0 &&
            sub.executions == sub.maxExecutions
        );
 
        // Cancel the subscription if no enough funds or subscription expired.
        // We do this after the execution so that the SubscriptionCancel event
        // is emitted after the SubscriptionExecuted event.
        if (
            !hasEnoughFunds || expired
        ) {
            _cancel(subscriptionHash, expired);
        }
    }

    /**
     * @dev Internal function that cancels a subscription.
     * @param subscriptionHash The hash of the subscription.
     */
    function _cancel(bytes32 subscriptionHash, bool expired) internal {
        address subscriber = _subscriptions[subscriptionHash].subscriber;
        require(
            subscriber != address(0),
            "Subscription doesn't exist"
        );
 
        // Remove from subscriber's array
        _removeFromArrayAndSwap(
            _subscriptionsBySubscriber[subscriber],
            _subscriptionSubscriberIndex,
            subscriptionHash
        );
 
        // Remove from all subscriptions array
        _removeFromArrayAndSwap(
            _allSubscriptions,
            _allSubscriptionsIndex,
            subscriptionHash
        );
 
        // Set all subscription fields to 0
        delete _subscriptions[subscriptionHash];
 
        _cancelled[subscriptionHash] = true;
 
        emit SubscriptionCancelled(subscriptionHash, expired);
    }

        address token,
        uint256 amount,
        PeriodUnit periodUnit,
        uint256 periods,
        uint256 maxExecutions,
        uint256 start
    )
        internal
    {
        require(
            _subscriptions[subscriptionHash].subscriber == address(0) &&
            !_cancelled[subscriptionHash],
            "Subscription exists"
        );
 
        require(
            subscriber != address(0),
            "Subscriber can't be the zero address"
        );
 
        require(periods != 0, "Periods can't be 0");
        require(amount != 0, "Amount can't be 0");
 
        // The subscription can't start in the past
        bool shouldStartNow = start <= now;
 
        uint256 startsAt = shouldStartNow ? now : start;
 
        _subscriptions[subscriptionHash] = Subscription(
            subscriber,
            token,
            amount,
            periodUnit,
            periods,
            maxExecutions,
            startsAt,
            0,
            now
        );
 
        _nextPayment[subscriptionHash] = startsAt;
 
        // Add subscription to subscriber's array
        bytes32[] storage subs = _subscriptionsBySubscriber[subscriber];
        _subscriptionSubscriberIndex[subscriptionHash] = subs.length;
        subs.push(subscriptionHash);
 
        // Add subscription to allSubscriptions array
        _allSubscriptionsIndex[subscriptionHash] = _allSubscriptions.length;
        _allSubscriptions.push(subscriptionHash);
 
        if (shouldStartNow) {
            execute(subscriptionHash);
        }
 
        emit SubscriptionCreated(subscriptionHash);
    }
 
    /**
     * @dev Internal helper for replay protection.
     * @param hash A hash that can only be used once.
     */
    function _replayGuard(bytes32 hash) internal {
        require(
            !_hashUsed[hash],
            "Hash already verified, use a different nonce"
        );
        _hashUsed[hash] = true;
    }
 
    /**
     * @dev Gets the hash of the string associated with a period unit.
     * We hash a string instead of the periodUnit value in order to provide
     * a better UX, as users can see the human readable unit when signing but
     * it maintains the same security guarantees.
     * @param periodUnit The period unit (enum)
     * @return The hash of the string that represents the given period unit.
     */
    function _getPeriodUnitHash(PeriodUnit periodUnit)
        internal
        pure
        returns(bytes32)
    {
        if (periodUnit == PeriodUnit.Day) {
            return keccak256("Day");
        } else if (periodUnit == PeriodUnit.Month) {
            return keccak256("Month");
        }
 
        return keccak256("Year");
    }
 
    /**
     * @dev Private function to remove a subscription from an array.
     * This has O(1) time complexity, but alters the order of the array.
     * @param array The array to remove the subscription from.
     * @param indexes A storage mapping that maps subscription hashes to
     * indexes.
     * @param item The subscription hash to remove.
     */
    function _removeFromArrayAndSwap(
        bytes32[] storage array,
        mapping(bytes32 => uint256) storage indexes,
        bytes32 item
    )
        private
    {
        // To prevent a gap in the array, we store the last item in the index
        // of the item to delete, and then delete the last slot (swap and pop).
        uint256 lastIndex = array.length.sub(1);
        uint256 index = indexes[item];
 
        // If the subscription being cancelled is the last in the array, this
        // is not necessary.
        if (index != lastIndex) {
            bytes32 lastItem = array[lastIndex];
 
            // Move the last item to the slot of the to-delete item
            array[index] = lastItem;
            // Update the moved item's index
            indexes[lastItem] = index;
        }
 
        // This also deletes the contents at the last position of the array
        array.length--;
        indexes[item] = 0;
    }
 
    /**
     * @dev Paginates a bytes32 array.
     * @param offset The offset in which the count begins.
     * @param limit Maximum amount of elements to return.
     */
    function _paginate(
        bytes32[] storage array,
        uint256 offset,
        uint256 limit
    )
        private
        view
        returns(bytes32[] memory items)
    {
        if (array.length <= offset) {
            return new bytes32[](0);
        }
 
        uint256 len = Math.min(limit, array.length - offset);
 
        items = new bytes32[](len);
 
        for (uint256 i = 0; i < len; i++) {
            items[i] = array[i + offset];
        }
    }
